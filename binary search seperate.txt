#include <iostream>
using namespace std;

struct Student {
    int id;
    string name;
    float cgpa;
};

int main() {
    int n = 5, sid, low=0, high=n-1, mid, found=0;
    Student* s = new Student[n];

    cout << "Enter 5 students (sorted by ID):\n";
    for(int i=0;i<n;i++)
        cin >> s[i].id >> s[i].name >> s[i].cgpa;

    cout << "\nEnter ID to search: ";
    cin >> sid;

    while(low <= high) {
        mid = (low + high)/2;
        if(s[mid].id == sid) {
            cout << "Found: " << s[mid].name << " " << s[mid].cgpa;
            found = 1; break;
        }
        (s[mid].id < sid) ? low = mid + 1 : high = mid - 1;
    }

    if(!found) cout << "Not Found";
    delete[] s;
}
