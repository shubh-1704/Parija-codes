#include <iostream>
using namespace std;

struct Student {
    int id;
    char name[50];
    float cgpa;
};

int main() {
    int n = 5;
    Student* s = new Student[n]; // dynamic memory allocation

    // Insertion of 5 records
    cout << "Enter details of 5 students:\n";
    for (int i = 0; i < n; i++) {
        cout << "\nStudent " << i + 1 << " ID: ";
        cin >> s[i].id;
        cout << "Name: ";
        cin >> s[i].name;
        cout << "CGPA: ";
        cin >> s[i].cgpa;
    }

    // Display all records
    cout << "\n--- All Student Records ---\n";
    for (int i = 0; i < n; i++)
        cout << s[i].id << "\t" << s[i].name << "\t" << s[i].cgpa << endl;

    // Search by ID (Linear Search)
    int searchID, found = 0;
    cout << "\nEnter ID to search: ";
    cin >> searchID;
    for (int i = 0; i < n; i++) {
        if (s[i].id == searchID) {
            cout << "Record Found: " << s[i].id << "\t" << s[i].name << "\t" << s[i].cgpa << endl;
            found = 1;
            break;
        }
    }
    if (!found)
        cout << "Record not found!\n";

    delete[] s; // free memory
    return 0;
}
