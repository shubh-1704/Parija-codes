#include <iostream>
using namespace std;

struct Student {
    int id;
    char name[50];
    float cgpa;
};

int main() {
    int n = 5;
    Student* s = new Student[n];

    cout << "Enter details of 5 students:\n";
    for (int i = 0; i < n; i++) {
        cout << "\nID: "; cin >> s[i].id;
        cout << "Name: "; cin >> s[i].name;
        cout << "CGPA: "; cin >> s[i].cgpa;
    }

    cout << "\n--- All Records ---\n";
    for (int i = 0; i < n; i++)
        cout << s[i].id << "\t" << s[i].name << "\t" << s[i].cgpa << endl;

    int id, found = 0;
    cout << "\nEnter ID to search: ";
    cin >> id;

    // Linear Search
    for (int i = 0; i < n; i++)
        if (s[i].id == id) { 
            cout << "Found by Linear Search: " << s[i].name << " (" << s[i].cgpa << ")\n"; 
            found = 1; break; 
        }
    if (!found) cout << "Not found by Linear Search!\n";

    // Binary Search (simple version â€” assume IDs entered in sorted order)
    int low = 0, high = n - 1, mid;
    found = 0;
    while (low <= high) {
        mid = (low + high) / 2;
        if (s[mid].id == id) { 
            cout << "Found by Binary Search: " << s[mid].name << " (" << s[mid].cgpa << ")\n"; 
            found = 1; break; 
        }
        else if (s[mid].id < id) low = mid + 1;
        else high = mid - 1;
    }
    if (!found) cout << "Not found by Binary Search!\n";

    delete[] s;
    return 0;
}
