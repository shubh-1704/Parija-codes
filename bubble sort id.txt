#include <iostream>
using namespace std;

struct Student {
    int id; string name; float cgpa;
};

int main() {
    int n=5;
    Student* s = new Student[n];
    for(int i=0;i<n;i++)
        cin >> s[i].id >> s[i].name >> s[i].cgpa;

    for(int i=0;i<n-1;i++)
        for(int j=0;j<n-i-1;j++)
            if(s[j].id > s[j+1].id)
                swap(s[j], s[j+1]);

    cout << "\nSorted by ID:\n";
    for(int i=0;i<n;i++)
        cout << s[i].id << " " << s[i].name << " " << s[i].cgpa << endl;

    delete[] s;
}
